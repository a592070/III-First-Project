create table account
(
	username varchar2(255) not null
		constraint ACCOUNT_PK
			primary key,
	password varchar2(255),
	register date
)

create table STOCKDAYS_ID
(
	day date not null
             	constraint STOCKDAYS_ID_PK
             		primary key,,
	trade_volume numeric(30),
	trade_value numeric(30),
	transaction numeric(30),
	hprice numeric(30,2),
	lprice numeric(30,2),
	openingprice numeric(30,2),
	closingprice numeric(30,2)
)
